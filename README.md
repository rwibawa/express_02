# express_02
nodeJS + expressJS exercise

1. Setup
  $  mkdir express_02
  $  cd express_02
  $  npm init
  $  npm install express
  $  touch app.js
  $  touch gulpfile.js
  $  vi app.js
  $  gulp --help
  $  vi gulpfile.js
  $  gulp --help
  $  node app.js

2. git
  $  git init
  $  git add -A
  $  git status
  $  git commit -m "Initialize repo"
  $  git log

  $  cat ~/.ssh/id_rsa.pub
  $  git remote add origin git@github.com:rwibawa/express_02.git
  $  git remote -v
  $  git push -u origin master


3. Heroku
  $  heroku login
  $  heroku create
  $  git push heroku master
  $  heroku logs --tail
